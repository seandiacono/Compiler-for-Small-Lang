ff XGreaterThanY(toCompare[]:int):auto{
    let ans:bool = false;
    if(toCompare[0] > toCompare[1]){
        ans = true;
    }

    return ans;
}

ff Average(toAverage[]:float, count:int):float{
    let total:float = 0.0;

    for(let i:int = 0; i<count; i = i+1){
        total = total + toAverage[i];
    }

    return total/count;
}
